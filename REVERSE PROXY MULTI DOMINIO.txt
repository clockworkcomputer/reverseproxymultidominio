ADMINISTRACÓN DE DOCKER

sudo docker ps

LISTAR CONTENEDORES ACTIVOS E INACTIVOS

docker stop $(docker ps -a -q)

docker rm $(docker ps -a -q)

docker rmi $(docker images -q)
